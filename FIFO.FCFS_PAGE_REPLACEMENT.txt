#include <stdio.h>

int main()
{
    int n,f,i,j,pages[50],frame[10];
    int hit=0,fault=0,front=0,found;

    printf("Enter number of pages: ");
    scanf("%d",&n);

    printf("Enter number of frames: ");
    scanf("%d",&f);

    printf("Enter page sequence: ");
    for(i=0;i<n;i++)
        scanf("%d",&pages[i]);

    for(i=0;i<f;i++)
        frame[i]=-1;

    printf("\nPage\t");
    for(i=0;i<f;i++)
        printf("F%d\t",i+1);
    printf("Status\n");

    for(i=0;i<n;i++)
    {
        found=0;

        for(j=0;j<f;j++)
            if(frame[j]==pages[i])
                found=1;

        if(found)
            hit++;
        else
        {
            frame[front]=pages[i];
            front=(front+1)%f;
            fault++;
        }

        printf("%d\t",pages[i]);
        for(j=0;j<f;j++)
            frame[j]==-1 ? printf("-\t") : printf("%d\t",frame[j]);

        found ? printf("Hit\n") : printf("Fault\n");
    }

    printf("\nTotal Hits   = %d",hit);
    printf("\nTotal Faults = %d",fault);
    printf("\nHit Ratio    = %.2f",(float)hit/n);
    printf("\nFault Ratio  = %.2f",(float)fault/n);

    return 0;
}
