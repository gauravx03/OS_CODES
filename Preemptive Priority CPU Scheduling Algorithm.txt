#include <stdio.h>

int main()
{
    int n,i,t=0,c=0,k=0;
    int at[20],bt[20],rt[20],ct[20],tat[20],wt[20],pr[20],done[20],g[200];

    printf("Enter number of processes: ");
    scanf("%d",&n);

    for(i=0;i<n;i++)
    {
        printf("AT BT PR of P%d: ",i+1);
        scanf("%d%d%d",&at[i],&bt[i],&pr[i]);
        rt[i]=bt[i];
        done[i]=0;
    }

    while(c<n)
    {
        int idx=-1;

        for(i=0;i<n;i++)
            if(!done[i] && at[i]<=t)
                if(idx==-1 || pr[i]<pr[idx] || (pr[i]==pr[idx] && rt[i]<rt[idx]))
                    idx=i;

        if(idx==-1)
        {
            g[k++]=-1;
            t++;
            continue;
        }

        g[k++]=idx;
        rt[idx]--;
        t++;

        if(rt[idx]==0)
        {
            ct[idx]=t;
            done[idx]=1;
            c++;
        }
    }

    float avgTAT=0,avgWT=0;

    printf("\nProcess\tAT\tBT\tPR\tCT\tTAT\tWT\n");

    for(i=0;i<n;i++)
    {
        tat[i]=ct[i]-at[i];
        wt[i]=tat[i]-bt[i];

        avgTAT+=tat[i];
        avgWT+=wt[i];

        printf("P%d\t%d\t%d\t%d\t%d\t%d\t%d\n",
        i+1,at[i],bt[i],pr[i],ct[i],tat[i],wt[i]);
    }

    printf("\nGantt Chart:\n|");
    for(i=0;i<k;i++)
        g[i]==-1 ? printf(" Idle |") : printf(" P%d |",g[i]+1);

    printf("\n0");
    for(i=1;i<=k;i++)
        printf("   %d",i);

    printf("\n\nAverage TAT = %.2f",avgTAT/n);
    printf("\nAverage WT  = %.2f",avgWT/n);

    return 0;
}
