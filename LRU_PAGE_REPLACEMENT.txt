#include <stdio.h>

int main()
{
    int n,f,i,j,pages[50],frame[10],time[10];
    int hit=0,fault=0,clk=0,pos,found;

    printf("Enter number of pages: ");
    scanf("%d",&n);

    printf("Enter number of frames: ");
    scanf("%d",&f);

    printf("Enter page sequence: ");
    for(i=0;i<n;i++)
        scanf("%d",&pages[i]);

    for(i=0;i<f;i++)
    {
        frame[i]=-1;
        time[i]=-1;
    }

    printf("\nPage\t");
    for(i=0;i<f;i++)
        printf("F%d\t",i+1);
    printf("Status\n");

    for(i=0;i<n;i++)
    {
        found=0;

        for(j=0;j<f;j++)
            if(frame[j]==pages[i])
            {
                found=1;
                time[j]=clk++;
            }

        if(found)
            hit++;
        else
        {
            pos=0;

            for(j=0;j<f;j++)
                if(frame[j]==-1)
                {
                    pos=j;
                    break;
                }

            if(frame[pos]!=-1)
                for(j=1;j<f;j++)
                    if(time[j]<time[pos])
                        pos=j;

            frame[pos]=pages[i];
            time[pos]=clk++;
            fault++;
        }

        printf("%d\t",pages[i]);
        for(j=0;j<f;j++)
            frame[j]==-1 ? printf("-\t") : printf("%d\t",frame[j]);

        found ? printf("Hit\n") : printf("Fault\n");
    }

    printf("\nTotal Hits   = %d",hit);
    printf("\nTotal Faults = %d",fault);
    printf("\nHit Ratio    = %.2f",(float)hit/n);
    printf("\nFault Ratio  = %.2f",(float)fault/n);

    return 0;
}
