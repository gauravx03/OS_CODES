#include <stdio.h>

int main()
{
    int n,i,j,t=0,temp;
    int at[20],bt[20],ct[20],tat[20],wt[20],p[20];

    printf("Enter number of processes: ");
    scanf("%d",&n);

    for(i=0;i<n;i++)
    {
        printf("AT BT of P%d: ",i+1);
        scanf("%d%d",&at[i],&bt[i]);
        p[i]=i+1;
    }

    for(i=0;i<n-1;i++)
        for(j=0;j<n-i-1;j++)
            if(at[j]>at[j+1])
            {
                temp=at[j]; at[j]=at[j+1]; at[j+1]=temp;
                temp=bt[j]; bt[j]=bt[j+1]; bt[j+1]=temp;
                temp=p[j];  p[j]=p[j+1];   p[j+1]=temp;
            }

    for(i=0;i<n;i++)
    {
        if(t<at[i]) t=at[i];
        t+=bt[i];
        ct[i]=t;
    }

    float a1=0,a2=0;

    printf("\nP\tAT\tBT\tCT\tTAT\tWT\n");

    for(i=0;i<n;i++)
    {
        tat[i]=ct[i]-at[i];
        wt[i]=tat[i]-bt[i];
        a1+=tat[i];
        a2+=wt[i];

        printf("P%d\t%d\t%d\t%d\t%d\t%d\n",
        p[i],at[i],bt[i],ct[i],tat[i],wt[i]);
    }

    printf("\nGantt Chart:\n|");
    t=0;

    for(i=0;i<n;i++)
    {
        if(t<at[i]) printf("Idle|"), t=at[i];
        printf("P%d|",p[i]);
        t+=bt[i];
    }

    printf("\n0");
    t=0;

    for(i=0;i<n;i++)
    {
        if(t<at[i]) t=at[i];
        t+=bt[i];
        printf(" %d",t);
    }

    printf("\n\nAverage TAT = %.2f",a1/n);
    printf("\nAverage WT  = %.2f",a2/n);

    return 0;
}
