#include <stdio.h>

int main()
{
    int n,i,t=0,c=0,k=0;
    int at[20],bt[20],ct[20],tat[20],wt[20],done[20],g[20];

    printf("Enter number of processes: ");
    scanf("%d",&n);

    for(i=0;i<n;i++)
    {
        printf("AT BT of P%d: ",i+1);
        scanf("%d%d",&at[i],&bt[i]);
        done[i]=0;
    }

    while(c<n)
    {
        int idx=-1;

        for(i=0;i<n;i++)
            if(!done[i] && at[i]<=t)
                if(idx==-1 || bt[i]<bt[idx] || (bt[i]==bt[idx] && at[i]<at[idx]))
                    idx=i;

        if(idx==-1) t++;
        else
        {
            g[k++]=idx;
            t+=bt[idx];
            ct[idx]=t;
            done[idx]=1;
            c++;
        }
    }

    float a1=0,a2=0;

    printf("\nP\tAT\tBT\tCT\tTAT\tWT\n");

    for(i=0;i<n;i++)
    {
        tat[i]=ct[i]-at[i];
        wt[i]=tat[i]-bt[i];
        a1+=tat[i];
        a2+=wt[i];

        printf("P%d\t%d\t%d\t%d\t%d\t%d\n",
        i+1,at[i],bt[i],ct[i],tat[i],wt[i]);
    }

    printf("\nGantt Chart:\n|");
    t=0;

    for(i=0;i<k;i++)
    {
        if(t<at[g[i]]) printf("Idle|"), t=at[g[i]];
        printf("P%d|",g[i]+1);
        t+=bt[g[i]];
    }

    printf("\n0");
    t=0;

    for(i=0;i<k;i++)
    {
        if(t<at[g[i]]) t=at[g[i]];
        t+=bt[g[i]];
        printf(" %d",t);
    }

    printf("\n\nAverage TAT = %.2f",a1/n);
    printf("\nAverage WT  = %.2f",a2/n);

    return 0;
}
