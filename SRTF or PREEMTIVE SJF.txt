#include <stdio.h>

int main()
{
    int n,i,t=0,c=0,k=0;
    int at[20],bt[20],rt[20],ct[20],tat[20],wt[20],g[200];

    printf("Enter number of processes: ");
    scanf("%d",&n);

    for(i=0;i<n;i++)
    {
        printf("AT BT of P%d: ",i+1);
        scanf("%d%d",&at[i],&bt[i]);
        rt[i]=bt[i];
    }

    while(c<n)
    {
        int idx=-1;

        for(i=0;i<n;i++)
            if(at[i]<=t && rt[i]>0)
                if(idx==-1 || rt[i]<rt[idx])
                    idx=i;

        if(idx==-1)
        {
            g[k++]=-1;
            t++;
        }
        else
        {
            g[k++]=idx;
            rt[idx]--;
            t++;

            if(rt[idx]==0)
            {
                ct[idx]=t;
                c++;
            }
        }
    }

    float a1=0,a2=0;

    printf("\nP\tAT\tBT\tCT\tTAT\tWT\n");

    for(i=0;i<n;i++)
    {
        tat[i]=ct[i]-at[i];
        wt[i]=tat[i]-bt[i];
        a1+=tat[i];
        a2+=wt[i];

        printf("P%d\t%d\t%d\t%d\t%d\t%d\n",
        i+1,at[i],bt[i],ct[i],tat[i],wt[i]);
    }

    printf("\nGantt Chart:\n|");
    for(i=0;i<k;i++)
        g[i]==-1 ? printf("Idle|") : printf("P%d|",g[i]+1);

    printf("\n0");
    for(i=1;i<=k;i++)
        printf(" %d",i);

    printf("\n\nAverage TAT = %.2f",a1/n);
    printf("\nAverage WT  = %.2f",a2/n);

    return 0;
}
