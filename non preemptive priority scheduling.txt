#include<stdio.h>

int main(){
    int n,i,time=0,completed=0,k=0;
    int at[20],bt[20],pr[20],ct[20],tat[20],wt[20],done[20],exec[20];

    printf("Enter number of processes: ");
    scanf("%d",&n);

    for(i=0;i<n;i++){
        printf("AT P%d: ",i+1);
        scanf("%d",&at[i]);

        printf("BT P%d: ",i+1);
        scanf("%d",&bt[i]);

        printf("Priority P%d: ",i+1);
        scanf("%d",&pr[i]);

        done[i]=0;
    }

    while(completed<n){
        int idx=-1;

        for(i=0;i<n;i++){
            if(!done[i] && at[i]<=time){
                if(idx==-1 || pr[i]<pr[idx])
                    idx=i;
            }
        }

        if(idx==-1){
            time++;
            continue;
        }

        exec[k++]=idx;
        time+=bt[idx];
        ct[idx]=time;

        done[idx]=1;
        completed++;
    }

    float avg_tat=0,avg_wt=0;

    for(i=0;i<n;i++){
        tat[i]=ct[i]-at[i];
        wt[i]=tat[i]-bt[i];
        avg_tat+=tat[i];
        avg_wt+=wt[i];
    }

    printf("\nProcess\tAT\tBT\tPR\tCT\tTAT\tWT\n");
    for(i=0;i<n;i++){
        printf("P%d\t%d\t%d\t%d\t%d\t%d\t%d\n",
        i+1,at[i],bt[i],pr[i],ct[i],tat[i],wt[i]);
    }

    // Gantt Chart
    printf("\n|");
    time=0;
    for(i=0;i<k;i++){
        if(time < at[exec[i]]){
            printf(" IDLE |");
            time = at[exec[i]];
        }
        printf(" P%d |",exec[i]+1);
        time += bt[exec[i]];
    }

    printf("\n0");
    time=0;
    for(i=0;i<k;i++){
        if(time < at[exec[i]]){
            time = at[exec[i]];
            printf(" %d",time);
        }
        time += bt[exec[i]];
        printf(" %d",time);
    }

    printf("\nAvg TAT=%.2f\nAvg WT=%.2f\n",avg_tat/n,avg_wt/n);
}
