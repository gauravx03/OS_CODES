#include <stdio.h>
#include <stdlib.h>

int main()
{
    int n, req[50], head, dir;
    int i, j, temp, current, total = 0;

    printf("Enter number of requests: ");
    scanf("%d", &n);

    printf("Enter requests: ");
    for(i=0; i<n; i++)
        scanf("%d", &req[i]);

    printf("Enter head position: ");
    scanf("%d", &head);

    printf("Enter direction (1=Right, 0=Left): ");
    scanf("%d", &dir);

    // Sort requests
    for(i=0; i<n-1; i++)
        for(j=0; j<n-i-1; j++)
            if(req[j] > req[j+1])
            {
                temp = req[j];
                req[j] = req[j+1];
                req[j+1] = temp;
            }

    current = head;

    printf("\nGantt Chart:\n");
    printf("%d", head);

    // Move Right
    if(dir == 1)
    {
        for(i=0; i<n; i++)
        {
            if(req[i] >= head)
            {
                total += abs(current - req[i]);
                current = req[i];
                printf(" -> %d", current);
            }
        }

        for(i=0; i<n; i++)
        {
            if(req[i] < head)
            {
                total += abs(current - req[i]);
                current = req[i];
                printf(" -> %d", current);
            }
        }
    }

    // Move Left
    else
    {
        for(i=n-1; i>=0; i--)
        {
            if(req[i] <= head)
            {
                total += abs(current - req[i]);
                current = req[i];
                printf(" -> %d", current);
            }
        }

        for(i=n-1; i>=0; i--)
        {
            if(req[i] > head)
            {
                total += abs(current - req[i]);
                current = req[i];
                printf(" -> %d", current);
            }
        }
    }

    printf("\n\nTotal Seek Length = %d", total);
    printf("\nAverage Seek Length = %.2f", (float)total/n);

    return 0;
}
